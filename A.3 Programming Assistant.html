data, info = load('ag_news_subset', with_info=True, as_supervised=True)

#data conversion
def preprocess_data(dataset):
    texts, labels = [], []
    for text, label in dataset:
        texts.append(text.numpy().decode('utf-8'))
        labels.append(label.numpy())
    return texts, np.array(labels)

train_texts, train_labels = preprocess_data(data['train'])
test_texts, test_labels = preprocess_data(data['test'])

#tokenization
tokenizer = tf.keras.preprocessing.text.Tokenizer(num_words=5000, oov_token='<OOV>')
tokenizer.fit_on_texts(train_texts)
train_sequences = tokenizer.texts_to_sequences(train_texts)
test_sequences = tokenizer.texts_to_sequences(test_texts)

MAX_LENGTH = 128  
train_padded = pad_sequences(train_sequences, maxlen=MAX_LENGTH, padding='post', truncating='post')
test_padded = pad_sequences(test_sequences, maxlen=MAX_LENGTH, padding='post', truncating='post')

#ohe
train_labels_cat = to_categorical(train_labels, num_classes=4)
test_labels_cat = to_categorical(test_labels, num_classes=4)

#split
X_train, X_val, y_train, y_val = train_test_split(train_padded, train_labels_cat, test_size=0.1, random_state=42)

#checking shapes b/c persisiting error

print("y_train shape:", y_train.shape)

def build_dense_model(dropout_rate=0.3, l2_reg=0.01):
    model = Sequential([
        Embedding(input_dim=5000, output_dim=64),
        Flatten(),
        Dense(128, activation='relu', kernel_regularizer=l2(l2_reg)),
        Dropout(dropout_rate),
        Dense(64, activation='relu', kernel_regularizer=l2(l2_reg)),
        Dropout(dropout_rate),
        Dense(4, activation='softmax')  
    ])
    model.compile(optimizer=Adam(), loss='categorical_crossentropy', metrics=['accuracy'])
    return model


def build_rnn_model(hidden_units=128, dropout_rate=0.3):
    model = Sequential([
        Embedding(input_dim=5000, output_dim=64),
        LSTM(hidden_units, return_sequences=False, dropout=dropout_rate),
        Dense(64, activation='relu'),
        Dense(4, activation='softmax')  
    ])
    model.compile(optimizer=RMSprop(), loss='categorical_crossentropy', metrics=['accuracy'])
    return model


def build_bidirectional_lstm_model(hidden_units=128, dropout_rate=0.3, optimizer=Adam()):
    model = Sequential([
        Embedding(input_dim=5000, output_dim=64),
        Bidirectional(LSTM(hidden_units, return_sequences=False, dropout=dropout_rate)),  
        Dense(64, activation='relu'),
        Dropout(dropout_rate),
        Dense(4, activation='softmax')  
    ])
    model.compile(optimizer=optimizer, loss='categorical_crossentropy', metrics=['accuracy'])
    return model


def build_cnn_model(kernel_size=5, dropout_rate=0.3, optimizer=SGD()):
    model = Sequential([
        Embedding(input_dim=5000, output_dim=64),
        Conv1D(128, kernel_size, activation='relu'),
        MaxPooling1D(pool_size=4),
        Flatten(),
        Dense(64, activation='relu'),
        Dropout(dropout_rate),
        Dense(4, activation='softmax')  
    ])
    model.compile(optimizer=optimizer, loss='categorical_crossentropy', metrics=['accuracy'])
    return model

def train_and_evaluate_model(model, X_train, y_train, X_val, y_val):
    sample_input = tf.random.uniform((5, 128), minval=0, maxval=5000, dtype=tf.int32)  
    sample_output = model(sample_input)
    print("Debugging: Model Output Shape (After Fix):", sample_output.shape)  
    assert sample_output.shape == (5, 4), "wrong"

    start_time = time.time()
    history = model.fit(X_train, y_train, epochs=10, batch_size=100, validation_data=(X_val, y_val))
    training_time = time.time() - start_time

    test_loss, test_acc = model.evaluate(test_padded, test_labels_cat)
    return history.history, test_loss, test_acc, training_time

def plot_model_performance(epochs, accuracy, loss, val_accuracy, val_loss, title):
    plt.figure(figsize=(12, 6))

    #acc
    plt.subplot(1, 2, 1)
    plt.plot(epochs, accuracy, label='Training Accuracy', marker = 'x')
    plt.plot(epochs, val_accuracy, label='Validation Accuracy', marker = 'o')
    plt.title(f'{title} - Accuracy')
    plt.xlabel('Epochs')
    plt.ylabel('Accuracy')
    plt.legend()

    #loss
    plt.subplot(1, 2, 2)
    plt.plot(epochs, loss, label='Training Loss', marker = 'x')
    plt.plot(epochs, val_loss, label='Validation Loss', marker = 'o')
    plt.title(f'{title} - Loss')
    plt.xlabel('Epochs')
    plt.ylabel('Loss')
    plt.legend()

    plt.tight_layout()
    plt.show()
